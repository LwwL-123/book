注意 new返回的是指针，而并没有实际的节点创建，&Struct是创建一个节点

```go
func addInList( head1 *ListNode ,  head2 *ListNode ) *ListNode {
    if head1 == nil && head2 == nil {
        return nil
    }
    
    head1 = reverse(head1)
    head2 = reverse(head2)
    
    res := &ListNode{0,nil}

    ifAdd := false
    for head1 != nil || head2 != nil || ifAdd {
        sum := 0
        if ifAdd {
            sum += 1
            ifAdd = false
        }
        
        if head1 != nil {
            sum += head1.Val
            head1 = head1.Next
        }
        
        if head2 != nil {
            sum += head2.Val
            head2 = head2.Next
        }
        
        if sum >= 10 {
            sum = sum - 10 
            ifAdd = true
        }
        
        node := &ListNode{sum,res}
        res = node
    }
    
  
    return res
}

func reverse(head *ListNode) *ListNode {
    if head == nil || head.Next == nil {
        return head
    }
    
    res := reverse(head.Next)
    
    head.Next.Next = head
    head.Next = nil
    
    return res
}
```
