```go
type LinkNode struct {
    key, val int
    pre, next *LinkNode
}

type LRUCache struct {
    m         map[int]*LinkNode
    cap       int
    head,tail *LinkNode
}


func Constructor(capacity int) LRUCache {
    head := &LinkNode{0,0,nil,nil}
    tail := &LinkNode{0,0,nil,nil}
    head.next = tail
    tail.pre = head

    return LRUCache{make(map[int]*LinkNode), capacity, head, tail}
}


func (this *LRUCache) Get(key int) int {
    cache := this.m

    if v,exist := cache[key]; exist {
        this.MoveToHead(v) 
        return v.val
    }else {
        return -1
    }
}


func (this *LRUCache) Put(key int, value int)  {
    head := this.head
    tail := this.tail
    cache := this.m

    if v,exist := cache[key]; exist {
        // 更新值
        v.val = value
        // 放到队头
        this.MoveToHead(v)
    }else {
        v := &LinkNode{key,value,nil,nil}
        if len(cache) == this.cap {
            delete(cache,tail.pre.key)
            tail.pre.pre.next = tail
            tail.pre = tail.pre.pre
        }
        // 插入新值
        v.next = head.next
        head.next.pre = v
        v.pre = head
        head.next = v
        cache[key] = v
    }

}

func (this *LRUCache) MoveToHead(node *LinkNode) {

    head := this.head

    // 从当前位置删除
    node.pre.next = node.next
    node.next.pre = node.pre

    // 移动到首位
    node.next = head.next
    head.next.pre = node
    node.pre = head
    head.next = node
}


/**
 * Your LRUCache object will be instantiated and called as such:
 * obj := Constructor(capacity);
 * param_1 := obj.Get(key);
 * obj.Put(key,value);
 */
```



#### [460. LFU 缓存](https://leetcode-cn.com/problems/lfu-cache/)

```go
type Node struct {
    key,val,freq int
    pre,next *Node
}

type DoubleList struct {
    head,tail *Node
}

type LFUCache struct {
    cap int
    keyToNode map[int]*Node
    freqToNodes map[int]*DoubleList
    minFreq int
}


func Constructor(capacity int) LFUCache {
    return LFUCache{
        cap: capacity,
        keyToNode: map[int]*Node{},
        freqToNodes: map[int]*DoubleList{},
        minFreq: 0,
    }
}


func (this *LFUCache) Get(key int) int {
    if node,exsit := this.keyToNode[key];exsit {
        // 使用频率+1
        this.increaseFreq(node)
        return node.val
    }
    return -1
}


func (this *LFUCache) Put(key int, value int)  {
    // 判断容量是否合法
    if this.cap <= 0 {
		return
	}
    // 判断有无
    if node,exsit := this.keyToNode[key]; exsit {
        // 如果存在，则更新值，增加频率
        node.val = value
        this.increaseFreq(node)
        return
    }

    if len(this.keyToNode) >= this.cap {
        // 删除频率最小的最近最久未使用
        this.deleteMinFreq()
    }

    //新增key
    node := &Node{
        key: key,
        val: value,
        freq: 1,
    }

    this.keyToNode[key] = node
    // 
    if this.freqToNodes[node.freq] == nil {
        this.freqToNodes[node.freq] = doubleListConstructor()
    }
    this.freqToNodes[node.freq].add(node)
    this.minFreq = 1
}

func (this *LFUCache) increaseFreq(node *Node) {
    // 从原先的双向队列中移除Node
    beforeFreq := node.freq
    oldDL := this.freqToNodes[beforeFreq]
    oldDL.remove(node)

    // 判断移除后是否为空，若为空minFreq++
    if oldDL.isEmpty() && beforeFreq == this.minFreq {
        this.minFreq++
    }

    // 使用频率+1
    node.freq++
    if this.freqToNodes[node.freq] == nil {
        this.freqToNodes[node.freq] = doubleListConstructor()
    }
    this.freqToNodes[node.freq].add(node)
}

func (this *LFUCache) deleteMinFreq() {
    dl := this.freqToNodes[this.minFreq]
    last := dl.tail.pre

    dl.remove(last)
    delete(this.keyToNode,last.key)
}


/**
 * Your LFUCache object will be instantiated and called as such:
 * obj := Constructor(capacity);
 * param_1 := obj.Get(key);
 * obj.Put(key,value);
 */

 func doubleListConstructor() *DoubleList {
	head, tail := &Node{}, &Node{}
	head.next, tail.pre = tail, head

	return &DoubleList{
		head: head,
		tail: tail,
	}
}

 func (this *DoubleList) add(node *Node) {
     h := this.head
     first := h.next

     h.next = node
     first.pre = node
     node.pre,node.next = h,first 
 }

 func (this *DoubleList) remove(node *Node) {
     node.pre.next = node.next
     node.next.pre = node.pre
     node.next = nil
     node.pre = nil
 }

 func (this *DoubleList) isEmpty() bool {
	return this.head.next == this.tail
}
```
